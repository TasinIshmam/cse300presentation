	\begin{frame}{Equivalence Property}
		\begin{itemize}
			\item We can also define UIS as a function of series of interactions, $i_i$\\
			\item We call 2 UIS equivalent if for every possible input, they produce the same result
		\end{itemize}
	\end{frame}
